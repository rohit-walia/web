Dynamic Pyramid

Rules:

Use only three characters: star *, forward-slash /, and backward-slash \

Should be perfectly symetrical

The tip of the pyramid should only consist of slashes and the base should only consist of stars. Everything in-between should have a combination of all three characters

Design the structure to be dynamic in size

You can't use nested for-loops

Solution via Java

 
                public class Pyramid {

                private static final char SLASH = '/';
                private static final char BACKSLASH ='\\';
                private static final char STAR = '*';
                private static int counter = 0;
                private static int width;
                private static final PrintStream write = System.out;

                public static void main(String[] args)
                {
                  printPyramid(12);
                }

                /*
                * Prints a pyramid structure given the height.
                *
                * @params height the height of the pyramid
                */
                public static void printPyramid(int height)
                {
                  counter++;
                  int numberOfStarsToPrint = height*2-2;

                  if(height==1)                                                         // base case
                  {
                    width = counter*2-2;
                    for (int i=0; i < width; i++)
                    {
                      if(i == Math.decrementExact(width/2))
                      {
                        write.print(SLASH);
                      }
                      else if(i == width/2)
                      {
                        write.print(BACKSLASH);
                      }
                      else
                      write.print(" ");
                    }
                    write.print(System.lineSeparator());
                    return;
                  }

                  printStarFigure(Math.decrementExact(height));                         // recursive step

                  int leftSide = Math.subtractExact(width/2, numberOfStarsToPrint/2);
                  int rightSide = Math.addExact(width/2, numberOfStarsToPrint/2);

                  for (int i=0; i < width; i++)
                  {
                    if(i < leftSide)
                    {
                      write.print(i == Math.decrementExact(leftSide) ? SLASH : " ");
                    }
                    else if(i >= rightSide)
                    {
                      write.print(i == rightSide ? BACKSLASH : " ");
                    }
                    else
                    write.print(STAR);
                  }
                  write.print(System.lineSeparator());
                  return;
                }
                }
